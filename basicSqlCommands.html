SQL Command Examples

GMT

Command 1: SELECT Statement

      
        SELECT * FROM customers;

Output:

customer_id customer_name email
1 John Smith john@example.com
2 Jane Doe jane@example.com
3 Mike Johnson mike@example.com

Command 2: UPDATE Statement

      
        
UPDATE employees SET salary = 50000 WHERE department = 'HR';

Output:

employee_id employee_name department salary
1 John Smith HR 50000
2 Jane Doe Sales 30000
3 Mike Johnson HR 50000

Command 3: INSERT Statement

      
        INSERT INTO orders (customer_id, product_name, quantity)
        VALUES (1, 'Product A', 5);

Output:

order_id customer_id product_name quantity
1 1 Product A 5

Command 4: DELETE Statement

      
        DELETE FROM products WHERE product_id = 3;

Output:

product_id product_name price
1 Product A 10
2 Product B 15

Command 5: SELECT DISTINCT Statement

  
    SELECT DISTINCT department FROM employees;

Output:

department
HR
Sales

Command 6: ORDER BY Statement

  
    SELECT * FROM customers ORDER BY customer_name ASC;

Output:

customer_id customer_name email
2 Jane Doe jane@example.com
3 Mike Johnson mike@example.com
1 John Smith john@example.com

Command 7: COUNT Function

  
    SELECT COUNT(*) AS total_customers FROM customers;

Output:

total_customers
3

Command 8: INNER JOIN

  
    SELECT orders.order_id, customers.customer_name, products.product_name
    FROM orders
    INNER JOIN customers ON orders.customer_id = customers.customer_id
    INNER JOIN products ON orders.product_id = products.product_id;

Output:

order_id customer_name product_name
1 John Smith Product A

Command 9: GROUP BY Statement

  
    SELECT department, COUNT(*) AS total_employees
    FROM employees
    GROUP BY department;

Output:

department total_employees
HR 2
Sales 1

Command 10: LIKE Operator

  
    SELECT * FROM products WHERE product_name LIKE '%A%';

Output:

product_id product_name price
1 Product A 10
